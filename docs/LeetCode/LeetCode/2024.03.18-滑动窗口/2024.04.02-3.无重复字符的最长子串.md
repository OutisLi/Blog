class Solution
{
public:
    int lengthOfLongestSubstring(string s)
    {
        unordered_map<char, int> window;
        int left = 0, right = 0;
        int res = 0;
        while (right < s.size())
        {
            char in = s[right];
            right++;
            window[in]++;
            while (window[in] > 1)
            {
                char out = s[left];
                left++;
                window[out]--;
            }
            res = max(res, right - left);
        }
        return res;
    }
};
